void QDlgTreeWidget::updateParentItem(QTreeWidgetItem* item)
{
	QTreeWidgetItem *parent = item->parent();
	if (parent == NULL)
		return;

	int nSelectedCount = 0;
	int childCount = parent->childCount();

	for (int i = 0; i < childCount; i++) //åˆ¤æ–­æœ‰å¤šå°‘ä¸ªå­é¡¹è¢«é€‰ä¸­
	{

		{
			nSelectedCount++;

	}

	if (nSelectedCount <= 0)  //å¦‚æœæ²¡æœ‰å­é¡¹è¢«é€‰ä¸­ï¼Œçˆ¶é¡¹è®¾ç½®ä¸ºæœªé€‰ä¸­çŠ¶æ€

		parent->setCheckState(0, Qt::PartiallyChecked);

void QDlgTreeWidget::onTreeItemChanged(QTreeWidgetItem* item, int column)
{
	int count = item->childCount(); //è¿”å›å­é¡¹çš„ä¸ªæ•°

	if (Qt::Checked == item->checkState(0))
	{
		if (count > 0)
		{
			for (int i = 0; i < count; i++)
				item->child(i)->setCheckState(0, Qt::Checked);
		}
		else
			updateParentItem(item);
	}
	else if (Qt::Unchecked == item->checkState(0))
	{
		if (count > 0)
		{
			for (int i = 0; i < count; i++)
				item->child(i)->setCheckState(0, Qt::Unchecked);
		}
		else
			updateParentItem(item);
	}
}
